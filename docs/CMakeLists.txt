set( DOXY_SOURCE_DIRECTORY ${PROJECT_SOURCE_DIR}/include )
set( DOXY_DOCS_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} )
set( DOXY_OUTPUT_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR} )

    VERBATIM
    SOURCES Doxyfile.in
)
